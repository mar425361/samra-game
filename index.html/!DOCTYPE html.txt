    const allData = {

          { "id": "op_12", "q": "من هو الأدميرال الأسطوري في البحرية والذي يُعتبر والد دراغون وجد لوفي؟", "a": "مونكي دي غارب" },
          { "id": "op_13", "q": "ما اسم السيف الأسطوري ذو اللون الأسود الذي كان يمتلكه ميهوك؟", "a": "يورو (Yoru)" }
        ],

          { "id": "bl_11", "q": "من هو قائد الفرقة العاشرة وأصغر قائد في تاريخ مجتمع الأرواح؟", "a": "هيتسوغايا التوشيرو" }
        ]
      },
      "مجموعة الجيمنج والألعاب 🎮": {
        "ألعاب السولز وإيلدن رينغ": [
          { "id": "sl_10", "q": "ما اسم المدينة الذهبية والعاصمة الرئيسية في لعبة إيلدن رينغ؟", "a": "ليندل (Leyndell)" },
          { "id": "sl_11", "q": "ما اسم الزعيم العملاق المقعد في إيلدن رينغ الذي يركب حصاناً صغيراً؟", "a": "جنرال رادان (Starscourge Radahn)" }
        ]
      }
    };

    let playedIds = [];
    let state = {
      question: "في انتظار بدء اللعبة من جوال المقدم... 🎲",
      answer: "",
      score1: 0,
      score2: 0,
      team1: "الفريق الأول",
      team2: "الفريق الثاني"
    };

    const urlParams = new URLSearchParams(window.location.search);
    const isAdmin = urlParams.get('mode') === 'admin';
    
    // إنشاء معرف غرف فريد تلقائيًا لمنع التضارب
    let roomId = urlParams.get('room');
    if (!roomId) {
      roomId = 'samra_' + Math.random().toString(36).substring(2, 8);
    }

    const syncUrl = `https://ntfy.sh/${roomId}/json`;

    window.onload = function() {
      if (isAdmin) {
        document.getElementById('tvView').style.display = 'none';
        document.getElementById('adminView').style.display = 'block';
        loadAdminGroups();
      } else {
        generateStaticQR();
        startTVListener();
      }
    };

    function generateStaticQR() {
      const baseUrl = window.location.href.split('?')[0];
      const adminUrl = `${baseUrl}?mode=admin&room=${roomId}`;
      new QRCode(document.getElementById("qrcode"), { text: adminUrl, width: 95, height: 95 });
    }

    function sendUpdate() {
      fetch(`https://ntfy.sh/${roomId}`, {
        method: 'POST',
        body: JSON.stringify(state)
      });
    }

    function startTVListener() {
      const eventSource = new EventSource(syncUrl);
      eventSource.onmessage = (event) => {
        try {
          const data = JSON.parse(event.data);
          const gameState = JSON.parse(data.message);
          
          document.getElementById('tvQuestion').innerText = gameState.question;
          document.getElementById('tvScore1').innerText = gameState.score1;
          document.getElementById('tvScore2').innerText = gameState.score2;
          document.getElementById('tvTeam1Name').innerText = gameState.team1;
          document.getElementById('tvTeam2Name').innerText = gameState.team2;
        } catch (e) {}
      };
    }

    function loadAdminGroups() {
      const gSelect = document.getElementById('adminGroupSelect');
      gSelect.innerHTML = '<option value="">-- اختر المجموعة --</option>';
      for (let g in allData) { gSelect.innerHTML += `<option value="${g}">${g}</option>`; }
    }

    function updateAdminCategories() {
      const g = document.getElementById('adminGroupSelect').value;
      const cSelect = document.getElementById('adminCategorySelect');
      cSelect.innerHTML = '<option value="">-- اختر التصنيف --</option>';
      if (g && allData[g]) { for (let c in allData[g]) { cSelect.innerHTML += `<option value="${c}">${c}</option>`; } }
    }

    function getRandomQuestion() {
      const g = document.getElementById('adminGroupSelect').value;
      const c = document.getElementById('adminCategorySelect').value;

      if (!g || !c) { alert("اختر المجموعة والتصنيف أولاً!"); return; }

      const available = allData[g][c].filter(q => !playedIds.includes(q.id));
      if (available.length === 0) { alert("انتهت الأسئلة لهذا التصنيف!"); return; }

      const q = available[Math.floor(Math.random() * available.length)];
      playedIds.push(q.id);

      state.question = q.q;
      state.answer = q.a;

      document.getElementById('adminQuestionBox').innerText = state.question;
      document.getElementById('adminAnswerBox').innerText = state.answer;

      sendUpdate();
    }

    function addScore(team, val) {
      if (team === 1) state.score1 = Math.max(0, state.score1 + val);
      else state.score2 = Math.max(0, state.score2 + val);

      document.getElementById('adminScore1').innerText = state.score1;
      document.getElementById('adminScore2').innerText = state.score2;

      sendUpdate();
    }

    function updateTeamNames() {
      state.team1 = document.getElementById('inputTeam1').value || "الفريق الأول";
      state.team2 = document.getElementById('inputTeam2').value || "الفريق الثاني";

      document.getElementById('labelTeam1').innerText = state.team1;
      document.getElementById('labelTeam2').innerText = state.team2;

      sendUpdate();
    }
