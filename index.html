`; const wheel = document.getElementById('spinningWheel'); // إعادة تصفير الدوران أولاً لضمان دورانها مع كل ضغطة wheel.style.transition = 'none'; wheel.style.transform = 'rotate(0deg)'; setTimeout(() => { wheel.style.transition = 'transform 3s cubic-bezier(0.15, 0.9, 0.2, 1)'; const randomDegree = Math.floor(1440 + Math.random() * 1080); wheel.style.transform = `rotate(${randomDegree}deg)`; }, 50); } else if (data.action === 'SHOW_CATEGORY') { // إبقاء العجلة ظاهرة لفترة قصيرة أو إخفائها وعرض الفئة المستقرة wheelContainer.style.display = 'none'; content.innerHTML = `

`; } else if (data.action === 'NOMINATE') { wheelContainer.style.display = 'none'; content.innerHTML = `

استعدوا للسؤال...
`; } else if (data.action === 'SHOW_Q') { wheelContainer.style.display = 'none'; content.innerHTML = `
